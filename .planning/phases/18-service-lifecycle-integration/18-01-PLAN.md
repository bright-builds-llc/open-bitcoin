```rust
use std::cell::RefCell;

#[derive(Debug, Clone, PartialEq, Eq)]
pub enum FakeServiceCall {
    Install { apply: bool },
    Uninstall { apply: bool },
    Enable,
    Disable,
    Status,
}

pub struct FakeServiceManager {
    pub recorded_calls: RefCell<Vec<FakeServiceCall>>,
    pub status_to_return: ServiceStateSnapshot,
    pub install_error: Option<ServiceError>,  // None = Ok; Some = return this error
}

impl FakeServiceManager {
    pub fn new(status_to_return: ServiceStateSnapshot) -> Self { ... }
}

impl ServiceManager for FakeServiceManager {
    fn install(...) -> Result<ServiceCommandOutcome, ServiceError> {
        self.recorded_calls.borrow_mut().push(FakeServiceCall::Install { apply: request.apply });
        if let Some(ref error) = self.install_error {
            return Err(error.clone()); // ServiceError must derive Clone
        }
        Ok(ServiceCommandOutcome { dry_run: !request.apply, description: "fake install".to_string(), ... })
    }
    // similarly for uninstall, enable, disable, status
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| subprocess output → Rust | `launchctl`/`systemctl` stdout/stderr parsed for state classification; malformed output must not panic |
| user-supplied binary_path → plist/unit generator | Injected into generated file; must be a real path, not validated for injection since it is operator-supplied and not an HTTP input |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-18-01 | Spoofing | subprocess output | mitigate | Parse exit code as primary signal; treat unexpected output as `ManagerCommandFailed` with stderr surfaced to operator; no secrets in stderr path |
| T-18-02 | Tampering | plist/unit file write | mitigate | Check AlreadyInstalled before write; --apply required; dry-run default prevents accidental overwrite |
| T-18-03 | Denial of service | double-install guard | mitigate | `ServiceError::AlreadyInstalled` returned when target file exists at apply time; no silent overwrite |
| T-18-04 | Information Disclosure | log_path in generated content | accept | Log paths are operator-visible filesystem paths; no credential values in generated file content |
| T-18-05 | Elevation of Privilege | user-level service scope | mitigate | User-scope by default (LaunchAgents / systemctl --user); no sudo required; system-scope deferred per D-10 |
</threat_model>
